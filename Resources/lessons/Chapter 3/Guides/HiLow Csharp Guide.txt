Chapter 3: HiLow
C# Guide

Prerequisites
	First we need to populate the file with the necessary class and main method:
		using System;
		public class HiLow
		{
			public static void Main(string[] args)
			{
			
			}
		}	

Create an answer between 1-100 and keep track of guesses
	Random rand = new Random();
	int answer = rand.Next(100) + 1;
	int numGuesses = 1;

Prompt the user to guess 1-100
	Console.WriteLine("Hello and welcome to the High-low number guessing game.");
	Console.Write("Guess a number between 1 and 100:");

Take in their input
	int guess;
	//this will make sure the input is an int, you could just use "guess = Console.ReadLine()" but this is more secure
	int.TryParse(Console.ReadLine(), out guess);

If they got it correct tell them how many guesses it took
	if (guess == answer)
	{
		Console.WriteLine($"Congrats you got it in {numGuesses} guesses.");
	}

Tell them if it's too high/low
	Continuing off of the previous if statement:
		else if(guess < answer)
		{
			Console.WriteLine("Too low");
			numGuesses++;
		}
		else if(guess > answer)
		{
			Console.WriteLine("Too high");
			numGuesses++;
		}

Allow them to keep guessing until it's correct
	This is where we wrap all this code in a do while loop
		int numGuesses = 1;
		bool win = false;
		do
		{
		...
		} while (!win);
	Don't forget to change the win variable:
		if (guess == answer)
		{
			win = true;
			...
		}

After that ask if they'd like to keep playing
	After the do while loop:
		Console.WriteLine("Would you like to play again?(y/n)");
		String playAgain = Console.ReadLine();
		if (playAgain.Equals("n"))
		{
			Console.WriteLine("Goodbye");
			playing = false;
		}
	And wrap up the whole thing in a while loop (or do/while):
		Console.WriteLine("Hello and welcome to the High-low number guessing game.");
		bool playing = true;
		while (playing)
		{
		...
		}
		
		
And that should do it, if you'd like to see the complete code look for HiLow.cs in the examples folder.
