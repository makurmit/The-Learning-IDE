Arrays and Arraylists
	//Remember arrays start at 0
	int[] nums = {10, 20, 30};
	//"=>" means returns this value
	//nums[0] => 10
	
	//foreach loops iterate through arrays and arraylists
	for(int num : nums)
	{
		System.out.print(num);
	}
	
	ArrayList nums = new ArrayList();
	nums.add(10);
	nums.remove(10);
	
	Python
		myList = [1, 2, 3]
		myList.append(4)
		del myList[1]
	
	Ruby
		myArr = [1, 2, 3]
		myArr.push(4)
		myArr.delete_at(1)
		
Enums
	public enum Name
	{
		value,
		otherValue,
		lastValue;
	}
	Name n = Name.value;
	
	Python
		from enum import Enum
		Name = Enum("Names", "value otherValue lastValue")
		Name.value
	
	Ruby
		module Name
			value = "a"
			otherValue = "b"
			lastValue = "c"
		end
		Name::value

Modulus
	a % b //returns remaining vale
	4 % 2 //returns 0
	5 % 2 //returns 1
